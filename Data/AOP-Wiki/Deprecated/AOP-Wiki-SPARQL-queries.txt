Endpoint : https://aopwiki.rdf.bigcat-bioinformatics.org/sparql/

1. Get list of AOs in AOP-Wiki: 

SELECT distinct ?AO ?AOtitle
WHERE {
 ?AOP a aopo:AdverseOutcomePathway ;
 aopo:has_adverse_outcome ?AO .
  ?AO dc:title ?AOtitle
}


2. Get KERs and corresponding KEs associated with AOPs that contain selected AOs:

SELECT DISTINCT ?KER ?KEup ?KEdown WHERE {
  ?AOP a aopo:AdverseOutcomePathway ;
  dc:title ?AOPTitle ; aopo:has_adverse_outcome ?AO;   aopo:has_key_event_relationship ?KER .
  ?KER aopo:has_upstream_key_event ?KEup ; aopo:has_downstream_key_event ?KEdown .
  VALUES ?AO { <https://identifiers.org/aop.events/1841> <https://identifiers.org/aop.events/352> }}


3. Get all KEs and their titles in AOP-Wiki

SELECT ?KE ?KEtitle
WHERE {
?KE a aopo:KeyEvent .
?KE dc:title ?KEtitle .
}


4. Get all KEs that have Ensembl gene ID associated with them and the Ensembl gene ID

SELECT  distinct  ?ke ?ensembl
WHERE {
	?ke a aopo:KeyEvent; edam:data_1025 ?object.
	?object skos:exactMatch ?id.
	?id a edam:data_1033; edam:data_1033 ?ensembl.
}

5. Get list of AOPs and AOs associated with selected KEs

SELECT DISTINCT ?AOP ?AOPTitle ?AO ?AOtitle WHERE {
  ?AOP a aopo:AdverseOutcomePathway ;
  dc:title ?AOPTitle ; aopo:has_adverse_outcome ?AO;   aopo:has_key_event ?KE.
  ?AO dc:title ?AOtitle
  VALUES ?KE { <https://identifiers.org/aop.events/279> <https://identifiers.org/aop.events/202> }}

Query results are downloaded in the .tsv format from the endpoint for smooth importing to Cytoscape
Before import, quotation marks are removed from the entire file
